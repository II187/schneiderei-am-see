Beispiel-Integration mit AI APIs:

// Beispiel: Stability AI API Integration
const generateClothing = async (prompt) => {
  const response = await fetch('https://api.stability.ai/v1/generation', {
    method: 'POST',
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      text_prompts: [{
        text: `Professional clothing photography: ${prompt}, 
               high quality, studio lighting, clean background`,
        weight: 1
      }],
      cfg_scale: 7,
      steps: 30,
      width: 512,
      height: 768
    })
  });
  
  const data = await response.json();
  return data.artifacts[0].base64;
};

// Alternative: DALL-E (OpenAI)
const generateWithDalle = async (prompt) => {
  const response = await fetch('https://api.openai.com/v1/images/generations', {
    method: 'POST',
    headers: {
      'Authorization': 'Bearer YOUR_OPENAI_KEY',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      prompt: `Fashion photography: ${prompt}`,
      n: 4,
      size: "512x768"
    })
  });
  
  const data = await response.json();
  return data.data.map(img => img.url);
};
